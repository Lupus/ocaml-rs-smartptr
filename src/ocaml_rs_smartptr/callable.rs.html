7/// The `Callable` trait represents a function or closure that can be called
8/// with a set of arguments to produce a return value. This trait is designed to
9/// be used with OCaml values and provides methods for calling the function,
10/// describing its arguments, and generating unique IDs for the function
11/// signature.
12///                                                                                                                                                                
13/// # Type Parameters                                                                                                                                              
14/// - `Ret`: The return type of the function, which must implement
15///   `ocaml::FromValue` and `OCamlDesc`.
16pub trait Callable<Ret>
17where
18    Ret: ocaml::FromValue + OCamlDesc,
19{
20    fn call_with(&self, gc: &ocaml::Runtime, func: ocaml::Value) -> Ret;
21    /// Describes the arguments (i.e. calls OCamlDesc::ocaml_desc) of the
22    /// function. This method should be provided by downstream trait
23    /// implementations.
24    fn describe_args(env: &::ocaml_gen::Env, generics: &[&str]) -> Vec<String>;
25    /// Generates unique IDs for the function arguments. This method should be
26    /// provided by downstream trait implementations.
27    fn unique_id_args() -> Vec<u128>;
28    /// ocaml_desc generates OCaml type signature for this Callable
29    fn ocaml_desc(env: &::ocaml_gen::Env, generics: &[&str]) -> String {
30        let args = Self::describe_args(env, generics)
31            .into_iter()
32            .map(|desc| format!("({})", desc))
33            .collect::<Vec<_>>()
34            .join(" -> ");
35        format!("({} -> ({}))", args, Ret::ocaml_desc(env, generics))
36    }
37
38    fn unique_id() -> u128 {
39        // Static randomized key for Callable
40        let key = highway::Key([
41            const_random!(u64),
42            const_random!(u64),
43            const_random!(u64),
44            const_random!(u64),
45        ]);
46        // Hasher seeded with our key
47        let mut hasher = HighwayHasher::new(key);
48        // Hash all Callable arguments
49        Self::unique_id_args()
50            .iter()
51            .for_each(|id| id.hash(&mut hasher));
52        /* Hash return value */
53        Ret::unique_id().hash(&mut hasher);
54        let result = hasher.finalize128();
55        (result[0] as u128) | ((result[1] as u128) << 64)
56    }
57    fn process_result(&self, res: Result<ocaml::Value, ocaml::Error>) -> Ret {
58        let res = res.unwrap();
59        Ret::from_value(res)
60    }
61}

63impl<Ret: ocaml::FromValue + OCamlDesc> Callable<Ret> for () {
64    fn call_with(&self, gc: &ocaml::Runtime, func: ocaml::Value) -> Ret {
65        // We use .call1 with a single `()' argument as OCaml does not have a
66        // notion of a function without arguments
67        self.process_result(unsafe { func.call1(gc, ()) })
68    }
69    fn describe_args(env: &ocaml_gen::Env, generics: &[&str]) -> Vec<String> {
70        // Just call OCamlDesc::ocaml_desc on `()' type
71        vec![<() as OCamlDesc>::ocaml_desc(env, generics)]
72    }
73    fn unique_id_args() -> Vec<u128> {
74        // Just call OCamlDesc::unique_id on `()' type
75        vec![<() as OCamlDesc>::unique_id()]
76    }
77}

79/// Macro to generate the `call_with` function for tuples of different sizes.
80/// This macro handles special cases for tuples with 1, 2, and 3 elements by
81/// generating the appropriate `func.call1`, `func.call2`, and `func.call3` calls.
82/// For tuples with more than 3 elements, it generates a generic `func.call`
83/// with the elements converted to OCaml values.
84macro_rules! generate_call_with {
85    ($idx:tt) => {
86        fn call_with(&self, gc: &ocaml::Runtime, func: ocaml::Value) -> Ret {
87            self.process_result(unsafe { func.call1(gc, &self.0) })
88        }
89    };
90    ($idx1:tt, $idx2:tt) => {
91        fn call_with(&self, gc: &ocaml::Runtime, func: ocaml::Value) -> Ret {
92            self.process_result(unsafe { func.call2(gc, &self.0, &self.1) })
93        }
94    };
95    ($idx1:tt, $idx2:tt, $idx3:tt) => {
96        fn call_with(&self, gc: &ocaml::Runtime, func: ocaml::Value) -> Ret {
97            self.process_result(unsafe { func.call3(gc, &self.0, &self.1, &self.2) })
98        }
99    };
100    ($count:tt, $($idx:tt),*) => {
101        fn call_with(&self, gc: &ocaml::Runtime, func: ocaml::Value) -> Ret {
102            self.process_result(unsafe {
103                func.call(
104                    gc,
105                    [
106                        $(
107                            self.$idx.to_value(gc),
108                        )*
109                    ],
110                )
111            })
112        }
113    };
114}

116/// Macro to implement the `Callable` trait for tuples of different sizes.
117/// This macro uses the `generate_call_with` macro to generate the `call_with`
118/// function and implements the `describe_args` and `unique_id_args` functions
119/// for tuples of different sizes.
120macro_rules! impl_callable_for_tuple {
121    ($($idx:literal),+) => {
122        paste! {
123            impl<
124                $(
125                    [<T $idx>]: ocaml::ToValue + OCamlDesc,
126                )*
127                Ret: ocaml::FromValue + OCamlDesc,
128            > Callable<Ret> for ($(
129                [<T $idx>],
130            )*)
131            {
132                generate_call_with! { $($idx),+ }
133                fn describe_args(env: &::ocaml_gen::Env, generics: &[&str]) -> Vec<String> {
134                    vec![
135                        $(
136                            [<T $idx>]::ocaml_desc(env, generics),
137                        )*
138                    ]
139                }
140                fn unique_id_args() -> Vec<u128> {
141                    vec![
142                        $(
143                            [<T $idx>]::unique_id(),
144                        )*
145                    ]
146                }
147            }
148        }
149    };
150}
