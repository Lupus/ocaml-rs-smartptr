1#![doc = r#"
2This module provides the `DynBox` smart pointer, which is a wrapper around the
3registry's `DynArc` with `PhantomData` for type safety. `DynBox` allows the user
4to wrap the object in a `Mutex` or shared `RwLock`. By default, using `.into()`
5will create a `Mutex`-protected version (exclusive).
6
7## Key Components
8
9- `DynBox<T>`: A smart pointer for dynamically typed Rust objects referenced
10  from the OCaml side.
11- `RustyObj`: A thin wrapper around a pointer to `DynArc`.
12
13## Usage
14
15### Creating a `DynBox`
16
17You can create a `DynBox` with either a `Mutex` or `RwLock`:
18
19```rust
20use ocaml_rs_smartptr::ptr::DynBox;
21let exclusive_box = DynBox::new_exclusive(42); // Mutex-protected
22let shared_box = DynBox::new_shared("foo"); // RwLock-protected
23```
24
25### Coercion
26
27The `coerce` and `coerce_mut` methods return a handle that holds a lock. Be
28cautious to avoid deadlocks when using these methods.
29
30### OCaml Integration
31
32`DynBox` integrates with the `ocaml_gen` package by providing `OCamlDesc` and
33`OCamlBinding` traits. Polymorphic variants are used to encode supported traits
34of the Rust type that we wrap/bind.
35
36Example:
37
38```ocaml
39module Animal = struct 
40  type nonrec t = [ `Ocaml_rs_smartptr_test_stubs_animal_proxy|`Core_marker_send ] Ocaml_rs_smartptr.Rusty_obj.t
41  external name : t -> string = "animal_name"
42  external noise : t -> string = "animal_noise"
43  external talk : t -> unit = "animal_talk"
44end
45
46
47module Sheep = struct 
48  type nonrec t = [ `Ocaml_rs_smartptr_test_stubs_sheep|`Core_marker_sync|`Core_marker_send|`Ocaml_rs_smartptr_test_stubs_animal_proxy ] Ocaml_rs_smartptr.Rusty_obj.t
49  external create : string -> t = "sheep_create"
50  external is_naked : t -> bool = "sheep_is_naked"
51  external sheer : t -> unit = "sheep_sheer"
52end
53
54
55module Wolf = struct 
56  type nonrec t = [ `Ocaml_rs_smartptr_test_stubs_wolf|`Core_marker_sync|`Core_marker_send|`Ocaml_rs_smartptr_test_stubs_animal_proxy ] Ocaml_rs_smartptr.Rusty_obj.t
57  external create : string -> t = "wolf_create"
58  external set_hungry : t -> bool -> unit = "wolf_set_hungry"
59end
60```
61
62This allows passing `Wolf` or `Sheep` whenever `Animal` is required by using
63coercion operator in OCaml (`:>`).

65### RustyObj
66
67`RustyObj` is a thin wrapper around a pointer to `DynArc`. We convert `Arc` into
68a raw pointer to hold that raw pointer in the OCaml heap, ensuring that moving
69of that value by the OCaml GC does not affect any Rust invariants. Reverse
70operation reconstructs the `Arc` from the raw pointer. This ensures that both
71OCaml and Rust always hold valid Arc-baked references to objects they need.
72"#]

84/// A smart pointer around the registry's `DynArc` with `PhantomData` for type safety.
85/// Allows the user to wrap the object in a `Mutex` or shared `RwLock`.
86/// By default, using `.into()` will create a `Mutex`-protected version (exclusive).
87pub struct DynBox<T>
88where
89    T: Send + ?Sized,
90{
91    inner: Arc<dyn Any + Sync + Send>,
92    _phantom: PhantomData<fn(T) -> T>, // https://doc.rust-lang.org/nomicon/phantom-data.html#table-of-phantomdata-patterns
93}
94
95impl<T: 'static + Send> DynBox<T> {
96    /// Creates a `DynBox` with a `Mutex`.
97    ///
98    /// # Parameters
99    ///
100    /// - `value`: The value to be wrapped in the `DynBox`.
101    ///
102    /// # Returns
103    ///
104    /// A new `DynBox` instance with `Mutex` protection.
105    pub fn new_exclusive(value: T) -> Self {
106        registry::register_type::<T>();
107        registry::register_type::<Arc<T>>();
108        DynBox {
109            inner: Arc::new(Mutex::new(value)),
110            _phantom: PhantomData,
111        }
112    }
113}
114
115impl<T: 'static + Send + ?Sized> DynBox<T> {
116    /// Creates a `DynBox` with a `Mutex` out of a Box'ed T. Useful if T is
117    /// unsized, e.g. a `dyn Trait`.
118    ///
119    /// # Parameters
120    ///
121    /// - `value`: The value (inside a Box) to be wrapped in the `DynBox`.
122    ///
123    /// # Returns
124    ///
125    /// A new `DynBox` instance with `Mutex` protection.
126    pub fn new_exclusive_boxed(value: Box<T>) -> Self {
127        registry::register_type::<Box<T>>();
128        registry::register_type::<Arc<Box<T>>>();
129        DynBox {
130            inner: Arc::new(Mutex::new(value)),
131            _phantom: PhantomData,
132        }
133    }
134}

177impl<T: 'static + Send + ?Sized> DynBox<T> {
178    fn into_raw(self) -> *const (dyn Any + Send + Sync) {
179        Arc::into_raw(self.inner)
180    }
181
182    fn from_raw(ptr: *const (dyn Any + Send + Sync)) -> Self {
183        DynBox {
184            inner: unsafe { Arc::from_raw(ptr) },
185            _phantom: PhantomData,
186        }
187    }
188
189    /// Coerces the `DynBox` to a handle of the specified type.
190    ///
191    /// # Returns
192    ///
193    /// A handle to the coerced type. Note that this handle holds a lock, so use
194    /// with care to avoid deadlocks.
195    pub fn coerce(&self) -> registry::Handle<T> {
196        registry::coerce::<T>(self.inner.clone())
197    }
198
199    /// Coerces the `DynBox` to a mutable handle of the specified type.
200    ///
201    /// # Returns
202    ///
203    /// A mutable handle to the coerced type. Note that this handle holds a
204    /// lock, so use with care to avoid deadlocks.
205    pub fn coerce_mut(&self) -> registry::HandleMut<T> {
206        registry::coerce_mut::<T>(self.inner.clone())
207    }
208}

210impl<T: 'static + Send + ?Sized> Clone for DynBox<T> {
211    fn clone(&self) -> Self {
212        DynBox {
213            inner: self.inner.clone(),
214            _phantom: PhantomData,
215        }
216    }

229impl<T: ?Sized + Send + 'static> OCamlDesc for DynBox<T> {
230    fn ocaml_desc(env: &::ocaml_gen::Env, _generics: &[&str]) -> String {
231        let type_id = <Self as OCamlDesc>::unique_id();
232        let typ = env
233            .get_type(type_id, type_name::get_type_name::<T>().as_str())
234            .0;
235        format!("_ {}'", typ)
236    }
237
238    fn unique_id() -> u128 {
239        let key = highway::Key([
240            const_random!(u64),
241            const_random!(u64),
242            const_random!(u64),
243            const_random!(u64),
244        ]);
245        let mut hasher = HighwayHasher::new(key);
246        let type_id = TypeId::of::<T>();
247        type_id.hash(&mut hasher);
248        let result = hasher.finalize128();
249        (result[0] as u128) | ((result[1] as u128) << 64)
250    }
251}

253impl<T: ?Sized + Send + 'static> OCamlBinding for DynBox<T> {
254    fn ocaml_binding(
255        env: &mut ::ocaml_gen::Env,
256        rename: Option<&'static str>,
257        new_type: bool,
258    ) -> String {
259        // register the new type
260        let ty_id = Self::unique_id();
261
262        if new_type {
263            let name = Box::leak(Box::new(type_name::get_type_name::<T>()));
264            let ty_name = rename.unwrap_or(name.as_str());
265            env.new_type(ty_id, ty_name);
266        }
267
268        let name = Self::ocaml_desc(env, &[]);
269        let name = name
270            .split_whitespace()
271            .last()
272            .expect("no last element :shrug:")
273            .to_owned();
274        let name = name
275            .strip_suffix("'")
276            .expect("dynbox type name does not end with `'`!");
277
278        let names = registry::get_type_info::<T>().implementations;
279        let variants = names
280            .iter()
281            .map(|type_str| type_name::snake_case_of_fully_qualified_name(type_str))
282            .map(|v| "`".to_owned() + &v)
283            .collect::<Vec<_>>()
284            .join("|");
285
286        if new_type {
287            format!(
288                "type tags = [{}] type 'a {}' = ([> tags ] as 'a) Ocaml_rs_smartptr.Rusty_obj.t type {} = tags {}'",
289                variants, name, name, name
290            )
291        } else {
292            let ty_name = rename.expect("bug in ocaml-gen: rename should be Some");
293            env.add_alias(ty_id, ty_name);
294
295            format!(
296                "type 'a {}' = 'a {}' type {} = {}",
297                ty_name, name, ty_name, name
298            )
299        }
300    }
301}

308/// A thin wrapper around a pointer to `DynArc`.
309/// We "leak" `Arc` into a raw pointer to hold that raw pointer in the OCaml
310/// heap, ensuring that moving of that value by the OCaml GC does not affect any
311/// Rust invariants.
312struct RustyObj(*const (dyn Any + Send + Sync));
313
314/// Finalizer is registered with OCaml GC, and ensures that our "leaked" `Arc`
315/// pointer is properly cleaned-up whenever OCaml drops corresponding object
316unsafe extern "C" fn rusty_obj_finalizer(v: ocaml::Raw) {
317    let ptr = v.as_pointer::<RustyObj>();
318    // Actual type parameter T for DynBox<T> is irrelevant here, dyn Any inside
319    // DynBox would know which destructor to call, and T is only for PhantomData
320    let dynbox: DynBox<i32> = DynBox::from_raw(ptr.as_ref().0);
321    drop(dynbox);
322    ptr.drop_in_place();
323}
324
325impl ocaml::Custom for RustyObj {
326    const NAME: &'static str = "RustyObj\0";
327
328    const OPS: ocaml::custom::CustomOps = ocaml::custom::CustomOps {
329        identifier: Self::NAME.as_ptr() as *mut ocaml::sys::Char,
330        finalize: Some(rusty_obj_finalizer),
331        ..ocaml::custom::DEFAULT_CUSTOM_OPS
332    };
333}

335unsafe impl<T> ocaml::FromValue for DynBox<T>
336where
337    T: Send + ?Sized + 'static,
338{
339    fn from_value(v: ocaml::Value) -> Self {
340        let ptr = unsafe { v.raw().as_pointer::<RustyObj>() };
341        let orig_dynbox = DynBox::from_raw(ptr.as_ref().0);
342        let dynbox = orig_dynbox.clone();
343        // orig_dynbox is owned by OCaml GC at this moment, so we can't drop it
344        // from Rust
345        let _ = std::mem::ManuallyDrop::new(orig_dynbox);
346        // dynbox is owned by Rust as a valid Arc clone, so we should be good to
347        // go to use it. Even if OCaml GC drops the original dynbox reference,
348        // we will proceed with our own
349        dynbox
350    }
351}
352
353unsafe impl<T> ocaml::ToValue for DynBox<T>
354where
355    T: Send + ?Sized + 'static,
356{
357    fn to_value(&self, rt: &ocaml::Runtime) -> ocaml::Value {
358        // Do a fresh clone of self and turn that into raw pointer
359        let ptr = DynBox::into_raw(self.clone());
360        // Convert to RustyObj to ensure that finalizer will be associated with
361        // raw Arc pointer
362        let rusty_obj = RustyObj(ptr);
363        ocaml::Pointer::from(rusty_obj).to_value(rt)
364    }
365}
