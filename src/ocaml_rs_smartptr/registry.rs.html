1//! This module provides a `Registry` for managing type coercions and type information
2//! in a Rust-OCaml binding context. The `Registry` allows for the registration of
3//! coercion functions that convert between different types, leveraging Rust's type
4//! system and OCaml's garbage collector.
5//!
6//! The primary goal is to facilitate the interaction between Rust and OCaml by
7//! providing a mechanism to register and retrieve type information and coercion
8//! functions. This is particularly useful when dealing with dynamic types (`dyn Any`)
9//! and ensuring that the OCaml garbage collector can properly manage Rust objects.
10//!
11//! The `Registry` uses `TypeId` to uniquely identify types and `DynArc` to store
12//! dynamically typed values. Coercion functions are stored in a `HashMap` and can be
13//! retrieved to convert between registered types.
14//!
15//! See relevant discussion: <https://users.rust-lang.org/t/rust-ocaml-bindings-and-traits/113263>
16//! Special thanks to Kevin Reid (<https://users.rust-lang.org/u/kpreid>) for
17//! providing the basis for building this module.
18

28/// An enum representing a read guard for either a `Mutex` or `RwLock`.
29/// This allows for a unified interface for read access to the underlying data.
30enum LockReadGuard<'a, T> {
31    Mutex(MutexGuard<'a, T>),
32    RwLockRead(RwLockReadGuard<'a, T>),
33}
34
35impl<T> Deref for LockReadGuard<'_, T> {
36    type Target = T;
37
38    fn deref(&self) -> &Self::Target {
39        match self {
40            LockReadGuard::Mutex(guard) => guard,
41            LockReadGuard::RwLockRead(guard) => guard,
42        }
43    }
44}
45
46/// Both MutexGuard and RwLockReadGuard are StableDeref, so enum of those two is
47/// also StableDeref
48unsafe impl<T> stable_deref_trait::StableDeref for LockReadGuard<'_, T> {}
49
50/// An enum representing a write guard for either a `Mutex` or `RwLock`.
51/// This allows for a unified interface for write access to the underlying data.
52enum LockWriteGuard<'a, T> {
53    Mutex(MutexGuard<'a, T>),
54    RwLockWrite(RwLockWriteGuard<'a, T>),
55}
56
57impl<T> Deref for LockWriteGuard<'_, T> {
58    type Target = T;
59
60    fn deref(&self) -> &Self::Target {
61        match self {
62            LockWriteGuard::Mutex(guard) => guard,
63            LockWriteGuard::RwLockWrite(guard) => guard,
64        }
65    }
66}
67
68impl<T> DerefMut for LockWriteGuard<'_, T> {
69    fn deref_mut(&mut self) -> &mut Self::Target {
70        match self {
71            LockWriteGuard::Mutex(guard) => &mut *guard,
72            LockWriteGuard::RwLockWrite(guard) => &mut *guard,
73        }
74    }
75}

78/// is also StableDeref
79unsafe impl<T> stable_deref_trait::StableDeref for LockWriteGuard<'_, T> {}

81/// A type alias for an `Arc` containing a dynamically typed value that is both
82/// `Sync` and `Send`. This is used to store values in the registry.
83type DynArc = Arc<dyn Any + Sync + Send>;
84
85/// Type alias for a function that takes a `DynArc` and returns a boxed `dyn Any`.
86/// This is used for type coercion in the registry.
87type CoercionInAny = Arc<dyn Fn(DynArc) -> Box<dyn Any> + Sync + Send>;
88
89/// A type alias for a handle to a read-only reference of type `Out`.
90/// This is used to represent coerced values in the registry.
91pub type Handle<Out> = ErasedBoxRef<Out>; // Holds a lock on DynArc
92
93/// A type alias for a handle to a mutable reference of type `Out`.
94/// This is used to represent coerced mutable values in the registry.
95pub type HandleMut<Out> = ErasedBoxRefMut<Out>; // Holds a lock on DynArc

97/// A struct representing type information, including the fully qualified name
98/// and a list of implementations.
99#[derive(Clone)]
100pub struct TypeInfo {
101    pub fq_name: &'static str,
102    pub implementations: Vec<&'static str>,
103}

105/// The `Registry` struct holds mappings for type coercions and type information.
106/// It allows registering coercion functions for converting between types and
107/// retrieving type information.
108#[derive(Default)]
109struct Registry {
110    traits: HashMap<(TypeId, TypeId), (CoercionInAny, CoercionInAny)>,
111    types: HashMap<TypeId, String>,
112    type_info_map: HashMap<TypeId, TypeInfo>,
113}
114
115impl Registry {
116    /// Creates a new `Registry` instance.
117    ///
118    /// # Returns
119    ///
120    /// A new `Registry` instance.
121    fn new() -> Self {
122        Self::default()
123    }
124
125    /// Registers coercion functions for converting between types `In` and `Out`.
126    ///
127    /// # Parameters
128    ///
129    /// - `fs`: A tuple containing two `CoercionInAny` functions for read and write coercions.
130    fn register_coercion_fns<In: Sized + 'static, Out: ?Sized + 'static>(
131        &mut self,
132        fs: (CoercionInAny, CoercionInAny),
133    ) {
134        self.traits
135            .insert((TypeId::of::<In>(), TypeId::of::<Out>()), fs);
136    }
137
138    /// Registers a type in the registry.
139    ///
140    /// # Parameters
141    ///
142    /// - `In`: The trait object type to register.
143    /// - `InReal`: The real type that implements the trait.
144    fn register_type<In: ?Sized + 'static, InReal: ?Sized>(&mut self) {
145        self.types
146            .insert(TypeId::of::<In>(), std::any::type_name::<InReal>().into());
147    }
148
149    /// Registers type information in the registry.
150    ///
151    /// # Parameters
152    ///
153    /// - `In`: The trait object type to register.
154    /// - `fq_name`: The fully qualified name of the type.
155    /// - `impls`: A vector of strings representing the implementations of the type.
156    fn register_type_info<In: ?Sized + 'static>(
157        &mut self,
158        fq_name: &'static str,
159        impls: Vec<&'static str>,
160    ) {
161        self.type_info_map.insert(
162            TypeId::of::<In>(),
163            TypeInfo {
164                fq_name,
165                implementations: impls,
166            },
167        );
168    }
169
170    /// Registers coercion functions for converting between types `In` and `Out`.
171    ///
172    /// # Parameters
173    ///
174    /// - `conv`: A function pointer for read coercion.
175    /// - `conv_mut`: A function pointer for write coercion.
176    fn register<In: Sized + 'static, Out: ?Sized + 'static>(
177        &mut self,
178        conv: fn(&In) -> &Out,
179        conv_mut: fn(&mut In) -> &mut Out,
180    ) {
181        // Retrieve the type name for the input type.
182        let type_in_name = String::from(self.type_name(&TypeId::of::<In>()));
183        // Clone the type name for use in the mutable coercion function.
184        let type_in_name_mut = type_in_name.clone();
185        // Create the read coercion function.
186        let f: CoercionInAny = Arc::new(move |boxed_t: DynArc| {
187            let ohandle = OwningHandle::new_with_fn(boxed_t, |bt| {
188                let any = unsafe { bt.as_ref() }.unwrap();
189                let guard = if let Some(mutex) = any.downcast_ref::<Mutex<In>>() {
190                    LockReadGuard::Mutex(mutex.lock().unwrap())
191                } else if let Some(rwlock) = any.downcast_ref::<RwLock<In>>() {
192                    LockReadGuard::RwLockRead(rwlock.read().unwrap())
193                } else {
194                    panic!(
195                        "unsupported container provided for coersion (type: {:?})",
196                        type_in_name
197                    );
198                };
199                OwningRef::new(guard).map(conv)
200            });
201            Box::new(OwningRef::new(ohandle).map_owner_box().erase_owner())
202        });
203        // Create the write coercion function.
204        let f_mut: CoercionInAny = Arc::new(move |boxed_t: DynArc| {
205            let ohandle = OwningHandle::new_with_fn(boxed_t, |bt| {
206                let any = unsafe { bt.as_ref() }.unwrap();
207                let guard = if let Some(mutex) = any.downcast_ref::<Mutex<In>>() {
208                    LockWriteGuard::Mutex(mutex.lock().unwrap())
209                } else if let Some(rwlock) = any.downcast_ref::<RwLock<In>>() {
210                    LockWriteGuard::RwLockWrite(rwlock.write().unwrap())
211                } else {
212                    panic!(
213                        "unsupported container provided for mut coersion (type: {:?})",
214                        type_in_name_mut
215                    );
216                };
217                OwningRefMut::new(guard).map_mut(conv_mut)
218            });
219            Box::new(OwningRefMut::new(ohandle).map_owner_box().erase_owner())
220        });
221        // Clone the coercion functions for registration.
222        let clone = || (f.clone(), f_mut.clone());
223        // Register the coercion functions for `Mutex<In>` to `Out`.
224        self.register_coercion_fns::<Mutex<In>, Out>(clone());
225        // Register the coercion functions for `RwLock<In>` to `Out`.
226        self.register_coercion_fns::<RwLock<In>, Out>(clone());
227    }
228
229    /// Retrieves the coercion functions for a given output type.
230    ///
231    /// # Parameters
232    ///
233    /// - `input`: A reference to a `DynArc` input.
234    ///
235    /// # Returns
236    ///
237    /// A tuple containing two `CoercionInAny` functions for read and write coercions.
238    fn get_coerce_fns<Out: ?Sized + 'static>(
239        &self,
240        input: &DynArc,
241    ) -> &(CoercionInAny, CoercionInAny) {
242        // Retrieve the `TypeId` of the input type.
243        // `**` is for: &Arc<dyn Any> -> Arc<dyn Any> -> dyn Any
244        let type_in = (**input).type_id();
245        // Retrieve the `TypeId` of the output type.
246        let type_out = TypeId::of::<Out>();
247        // Retrieve the type name for the input type.
248        let type_in_name = self.type_name(&type_in);
249        // Retrieve the coercion functions from the registry.
250        self.traits.get(&(type_in, type_out)).unwrap_or_else(|| {
251            panic!(
252                "there is no registered coercion for {:?} => {:?}",
253                type_in_name,
254                std::any::type_name::<Out>()
255            )
256        })
257    }
258
259    /// Retrieves the type name for a given `TypeId`.
260    ///
261    /// # Parameters
262    ///
263    /// - `type_in`: A reference to a `TypeId`.
264    ///
265    /// # Returns
266    ///
267    /// A string slice representing the type name.
268    fn type_name(&self, type_in: &TypeId) -> &str {
269        // Retrieve the type name from the registry.
270        match self.types.get(type_in) {
271            Some(name) => name.as_str(),
272            None => "<unregistered type>",
273        }
274    }
275
276    /// Coerces a `DynArc` input to a handle of the specified output type.
277    ///
278    /// # Parameters
279    ///
280    /// - `input`: A `DynArc` input.
281    ///
282    /// # Returns
283    ///
284    /// A handle to the coerced output type.
285    fn coerce<Out: ?Sized + 'static>(&self, input: DynArc) -> Handle<Out> {
286        // Retrieve the read coercion function.
287        let (f, _) = self.get_coerce_fns::<Out>(&input);
288        // Coerce the input to the output type.
289        // Coerce the input to the mutable output type.
290        *f(input.clone())
291            .downcast()
292            .expect("coercion fn returned wrong type")
293    }
294
295    /// Coerces a `DynArc` input to a mutable handle of the specified output type.
296    ///
297    /// # Parameters
298    ///
299    /// - `input`: A `DynArc` input.
300    ///
301    /// # Returns
302    ///
303    /// A mutable handle to the coerced output type.
304    fn coerce_mut<Out: ?Sized + 'static>(&self, input: DynArc) -> HandleMut<Out> {
305        // Retrieve the write coercion function.
306        let (_, f) = self.get_coerce_fns::<Out>(&input);
307        *f(input.clone())
308            .downcast()
309            .expect("coercion fn returned wrong type")
310    }
311
312    /// Retrieves the type information for a given input type.
313    ///
314    /// # Parameters
315    ///
316    /// - `In`: The trait object type to retrieve information for.
317    ///
318    /// # Returns
319    ///
320    /// A `TypeInfo` struct containing the type information.
321    fn get_type_info<In: ?Sized + 'static>(&self) -> TypeInfo {
322        // Retrieve the `TypeId` of the input type.
323        let type_id = TypeId::of::<In>();
324        // Retrieve the type information from the registry.
325        let type_info = self.type_info_map.get(&type_id).unwrap_or_else(|| {
326            panic!(
327                "registry does not contain a registered type info for {}",
328                std::any::type_name::<In>()
329            )
330        });
331        type_info.clone()
332    }
333}

335/// Returns a reference to the global registry.
336///
337/// # Returns
338///
339/// A reference to the global `RwLock<Registry>`.
340fn global_registry() -> &'static RwLock<Registry> {
341    // Initialize the global registry.
342    static REGISTRY: OnceLock<RwLock<Registry>> = OnceLock::new();
343    REGISTRY.get_or_init(|| RwLock::new(Registry::new()))
344}

346/// Registers coercion functions for converting between types `In` and `Out` in the global registry.
347///
348/// # Parameters
349///
350/// - `conv`: A function pointer for read coercion.
351/// - `conv_mut`: A function pointer for write coercion.
352pub fn register<In: Sized + 'static, Out: ?Sized + 'static>(
353    conv: fn(&In) -> &Out,
354    conv_mut: fn(&mut In) -> &mut Out,
355) {
356    // Obtain a write lock on the global registry.
357    let mut registry = global_registry()
358        .write()
359        .expect("unable to obtain write lock on global registry");
360    registry.register::<In, Out>(conv, conv_mut)
361}
362
363/// Registers a type in the global registry.
364///
365/// # Parameters
366///
367/// - `In`: The trait object type to register.
368pub fn register_type<In: ?Sized + 'static>() {
369    let mut registry = global_registry()
370        .write()
371        .expect("unable to obtain write lock on global registry");
372    registry.register_type::<In, In>();
373    registry.register_type::<Mutex<In>, In>();
374    registry.register_type::<RwLock<In>, In>();
375}

394/// Coerces a `DynArc` input to a handle of the specified output type using the global registry.
395///
396/// # Parameters
397///
398/// - `input`: A `DynArc` input.
399///
400/// # Returns
401///
402/// A handle to the coerced output type.
403pub fn coerce<Out: ?Sized + 'static>(input: DynArc) -> Handle<Out> {
404    // Note: This function holds a lock on DynArc. Use with care to avoid deadlocks.
405    // Obtain a read lock on the global registry.
406    let registry = global_registry()

445/// The `Plugin` struct represents a plugin with an initializer function.
446pub struct Plugin {
447    /// A function pointer to the initializer function.
448    initializer: fn(),
449}
450
451impl Plugin {
452    /// Creates a new `Plugin` with the given initializer function.
453    ///
454    /// # Parameters
455    ///
456    /// - `initializer`: A function pointer to the initializer function.
457    ///
458    /// # Returns
459    ///
460    /// A new `Plugin` instance.
461    pub const fn new(initializer: fn()) -> Self {
462        // Create a new `Plugin` instance with the given initializer function.
463        Plugin { initializer }
464    }
465
466    /// Initializes the plugin by calling its initializer function.
467    fn initialize(&self) {
468        // Call the initializer function.
469        (self.initializer)();
470    }
471}

477/// Initializes all registered plugins. This function is called once.
478pub fn initialize_plugins() {
479    // Initialize all registered plugins.
480    INIT.call_once(|| {
481        for plugin in inventory::iter::<Plugin> {
482            plugin.initialize();
483        }
