3use crate::callable::Callable;
4use crate::ml_box::MlBox;
5use std::marker::PhantomData;
6use std::panic::{AssertUnwindSafe, RefUnwindSafe, UnwindSafe};

8/// OCamlFunc is a wrapper around MlBox that represents an OCaml function.                                                                                        
9/// It holds a reference to the OCaml function and ensures that it is safe to call                                                                                
10/// from Rust. The PhantomData is used to keep track of the argument and return types.
11#[derive(Debug)]
12pub struct OCamlFunc<Args, Ret>(MlBox, AssertUnwindSafe<PhantomData<(Args, Ret)>>);
13
14// As OCamlFunc is a wraper on top of MlBox, we mark OCamlFunc as Send + Sync as
15// MlBox itself
16unsafe impl<Args, Ret> Send for OCamlFunc<Args, Ret> {}
17unsafe impl<Args, Ret> Sync for OCamlFunc<Args, Ret> {}
18
19assert_impl_all!(OCamlFunc<(ocaml::Value,),ocaml::Value>: Send, Sync, UnwindSafe, RefUnwindSafe);
20
21impl<Args, Ret> OCamlFunc<Args, Ret> {
22    /// Creates a new OCamlFunc from an OCaml value.                                                                                                              
23    /// This function takes an OCaml runtime handle to ensure that the operation                                                                                  
24    /// is called while the OCaml domain lock is acquired.
25    pub fn new(gc: &ocaml::Runtime, v: ocaml::Value) -> Self {
26        OCamlFunc(MlBox::new(gc, v), AssertUnwindSafe(PhantomData))
27    }
28}
29
30impl<Args, Ret> Clone for OCamlFunc<Args, Ret> {
31    /// Clones the OCamlFunc, creating a new instance with the same underlying OCaml function.
32    /// Custom Clone implementation lifts the requirements for Args and Ret to be Clone
33    fn clone(&self) -> Self {
34        OCamlFunc(self.0.clone(), AssertUnwindSafe(PhantomData))
35    }
36}
37
38unsafe impl<Args, Ret> ocaml::FromValue for OCamlFunc<Args, Ret> {
39    /// Converts an OCaml value to an OCamlFunc.                                                                                                                  
40    /// This function should ideally receive a runtime handle, but it assumes that                                                                                
41    /// it is not called manually on a non-OCaml thread.
42    fn from_value(v: ocaml::Value) -> Self {
43        OCamlFunc::new(unsafe { ocaml::Runtime::recover_handle() }, v)
44    }
45}
46
47impl<Args: Callable<Ret>, Ret: ocaml::FromValue> OCamlFunc<Args, Ret>
48where
49    Ret: OCamlDesc,
50{
51    /// Calls the OCaml function with the provided arguments.                                                                                                     
52    /// This function ensures that the OCaml runtime is properly handled.
53    pub fn call(&self, gc: &ocaml::Runtime, args: Args) -> Ret {
54        args.call_with(gc, self.0.as_value(gc))
55    }
56}

58/// OCamlDesc impl for OCamlFunc is a thin wrapper on top of corresponding
59/// methods in Callable.
60impl<Args, Ret> OCamlDesc for OCamlFunc<Args, Ret>
61where
62    Args: Callable<Ret>,
63    Ret: ocaml::FromValue + OCamlDesc,
64{
65    /// Generates the OCaml type description for the function.
66    fn ocaml_desc(env: &::ocaml_gen::Env, generics: &[&str]) -> String {
67        Args::ocaml_desc(env, generics)
68    }
69
70    /// Generates a unique ID for the function.
71    fn unique_id() -> u128 {
72        Args::unique_id()
73    }
