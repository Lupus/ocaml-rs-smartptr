1//! This module provides a safe wrapper around `ocaml::Value` to allow
2//! sending OCaml values between threads in Rust. The `MlBox` struct ensures
3//! that the OCaml runtime handle is available when recovering the value,
4//! maintaining safety.
5
6use std::panic::{AssertUnwindSafe, RefUnwindSafe, UnwindSafe};
7use std::sync::Arc;
8
9use derive_more::derive::Display;
10
11/// This wrapper type around `ocaml::root::Root` allows sending
12/// `ocaml::root::Root` to other threads and provides a safe interface for doing
13/// so - the value can be recovered back only in a context where an OCaml
14/// runtime handle is available.  As cloning of `ocaml::root::Root` is not safe
15/// outside of the OCaml Domain lock (see comments in the `as_value` method
16/// below), we wrap it with `Arc` to enable safe cloning of `MlBox` from Rust.
17///
18/// `ocaml::root::Root` is a wrapper type around what's returned by the `boxroot_create()`
19/// C function. For more details, refer to the [boxroot documentation](https://gitlab.com/ocaml-rust/ocaml-boxroot/-/blob/main/boxroot/boxroot.h).
20#[derive(Clone, Debug, Display)]
21#[display("MlBox<{:?}>", inner)]
22pub struct MlBox {
23    inner: Arc<AssertUnwindSafe<ocaml::root::Root>>,
24}
25
26/// The box root is just a pointer, wrapped by `Arc`, so `MlBox` is thus safe to send to
27/// other threads.
28unsafe impl Send for MlBox {}
29/// All methods of `MlBox` require the OCaml runtime, and thus cannot be concurrently
30/// run from different threads, making `MlBox` Sync. The only exception is `.clone()`,
31/// but that is handled by `Arc`, so it's perfectly Sync too.
32unsafe impl Sync for MlBox {}
33
34assert_impl_all!(MlBox: Send, Sync, UnwindSafe, RefUnwindSafe);
35
36impl MlBox {
37    /// Creates a new `MlBox` out of `ocaml::Value`, taking an OCaml runtime handle to
38    /// ensure this operation is called while the OCaml domain lock is acquired.
39    pub fn new(_gc: &ocaml::Runtime, value: ocaml::Value) -> Self {
40        match value {
41            ocaml::Value::Raw(v) => {
42                // `ocaml::Value` was a raw one, need to create a new root for it
43                // to avoid it from being garbage collected by the OCaml GC.
44                Self {
45                    #[allow(clippy::arc_with_non_send_sync)]
46                    inner: Arc::new(AssertUnwindSafe(unsafe {
47                        ocaml::root::Root::new(v)
48                    })),
49                }
50            }
51            ocaml::Value::Root(r) => {
52                // `ocaml::Value` was already rooted, so we can just take the root
53                // out of it and safely proceed with it further.
54                Self {
55                    #[allow(clippy::arc_with_non_send_sync)]
56                    inner: Arc::new(AssertUnwindSafe(r)),
57                }
58            }
59        }
60    }
61
62    /// Consumes this `MlBox` to recover the original `ocaml::Value` (it will be a
63    /// rooted one) if the internal `Arc` was the only strong reference, otherwise
64    /// returns `None`. Generally, using `as_value` is more convenient. This method
65    /// can be used when you're sure that you have only one reference to `MlBox`,
66    /// in this case using this method can save on new boxroot allocation.
67    pub fn into_value(self, _gc: &ocaml::Runtime) -> Option<ocaml::Value> {
68        Arc::into_inner(self.inner)
69            .map(|x| x.0)
70            .map(ocaml::Value::Root)
71    }
72
73    /// Creates a new rooted `ocaml::Value`, the root is obtained by recovering the value
74    /// from the current root and creating a new root for it.
75    pub fn as_value(&self, _gc: &ocaml::Runtime) -> ocaml::Value {
76        // Caveat: we call `.clone()` on `ocaml::root::Root`, which will create a
77        // new boxroot with the value obtained from the current boxroot. According to
78        // `boxroot.h`, both `boxroot_create` and `boxroot_get` require the OCaml
79        // domain lock to be held by the current thread. I.e. calling `.clone()` on
80        // `ocaml::root::Root` is generally not safe to do unless the OCaml domain
81        // lock is held - hence this function ensures that we have an OCaml runtime
82        // reference to maintain the safety guarantees.
83        let AssertUnwindSafe(new_root): &AssertUnwindSafe<ocaml::root::Root> =
84            self.inner.as_ref();
85        ocaml::Value::Root(new_root.clone())
86    }
87}
88
89unsafe impl ocaml::ToValue for MlBox {
90    fn to_value(&self, gc: &ocaml::Runtime) -> ocaml::Value {
91        self.as_value(gc)
92    }
93}
